def notify(msg)
  # Case 1: null or empty string
  if flash[:alert].nil? || flash[:alert].to_s.strip.length == 0
    flash[:alert] = msg
  # Case 2: array
  elsif flash[:alert].kind_of?(Array)
    flash[:alert].push(msg)
  # Case 3: contains a string
  else
    str = flash[:alert]
    flash[:alert] = [str, msg]
  end
end
